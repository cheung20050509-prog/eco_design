# ========== 海龟交易法核心指标计算 ==========
def calculate_technical_indicators(data, atr_period=20, entry_period=20, exit_period=10):
    """
    计算海龟交易法所需的技术指标
    
    参数:
    data: DataFrame，包含OHLC数据
    atr_period: ATR计算周期，默认20日
    entry_period: 入场突破周期，默认20日
    exit_period: 出场突破周期，默认10日
    
    返回:
    添加了技术指标的DataFrame
    """
    # 计算真实波幅(TR)
    # TR = Max(当日最高价-当日最低价, |当日最高价-前收盘价|, |当日最低价-前收盘价|)
    data['high_low'] = data['high'] - data['low']
    data['high_prev_close'] = abs(data['high'] - data['close'].shift(1))
    data['low_prev_close'] = abs(data['low'] - data['close'].shift(1))
    data['TR'] = data[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # 计算ATR（简单移动平均方法）
    # ATR = ((前1日ATR×(n-1)) + 当日TR)/n
    data['ATR'] = data['TR'].rolling(window=atr_period).mean()
    
    # 计算入场和出场信号
    # 入场：突破N日最高价（唐奇安通道上轨）
    data['entry_signal'] = data['high'].rolling(window=entry_period).max().shift(1)
    # 出场：跌破M日最低价（唐奇安通道下轨）
    data['exit_signal'] = data['low'].rolling(window=exit_period).min().shift(1)
    
    # 计算20日移动平均线（趋势参考）
    data['MA20'] = data['close'].rolling(window=20).mean()
    
    # 计算波动率（用于头寸规模计算）
    data['volatility'] = data['close'].rolling(window=20).std() / data['close'].rolling(window=20).mean()
    
    # 清理临时列
    data.drop(['high_low', 'high_prev_close', 'low_prev_close'], axis=1, inplace=True)
    
    return data

# 计算技术指标
df = calculate_technical_indicators(df)

# ========== 海龟交易法策略实现 ==========
class TurtleTradingStrategy:
    """
    简化的海龟交易策略实现
    """
    def __init__(self, initial_capital=1000000, risk_per_trade=0.01, atr_multiplier=2):
        """
        初始化策略参数
        
        参数:
        initial_capital: 初始资金，默认100万
        risk_per_trade: 每笔交易风险，默认1%
        atr_multiplier: ATR倍数，用于计算止损，默认2倍ATR
        """
        self.initial_capital = initial_capital
        self.capital = initial_capital
        self.risk_per_trade = risk_per_trade
        self.atr_multiplier = atr_multiplier
        self.position = 0  # 持仓数量
        self.entry_price = 0  # 入场价格
        self.stop_loss = 0  # 止损价格
        self.trades = []  # 记录交易
        self.equity_curve = []  # 记录权益曲线
        
    def calculate_position_size(self, price, atr):
        """
        计算头寸规模
        
        公式:
        头寸单位 = (账户净值 × 风险百分比) / (ATR倍数 × ATR)
        
        注意：假设单点价值为1（即股价每变动1元，盈亏变动1元）
        """
        if atr == 0 or price == 0:
            return 0
            
        # 计算风险资金
        risk_amount = self.capital * self.risk_per_trade
        
        # 计算头寸单位（简化为股数）
        position_units = risk_amount / (self.atr_multiplier * atr * price)
        
        # 取整（股票交易最小单位100股）
        position_units = int(position_units / 100) * 100
        
        return position_units
    
    def execute_trades(self, data):
        """
        执行交易策略
        """
        in_position = False
        self.equity_curve = []
        
        for i in range(20, len(data)):
            current_date = data.loc[i, 'trade_date']
            current_price = data.loc[i, 'close']
            atr = data.loc[i, 'ATR']
            entry_signal = data.loc[i, 'entry_signal']
            exit_signal = data.loc[i, 'exit_signal']
            
            # 更新权益曲线（每日）
            daily_equity = self.capital + self.position * (current_price - self.entry_price) if in_position else self.capital
            self.equity_curve.append({
                'date': current_date,
                'equity': daily_equity,
                'price': current_price
            })
            
            if not in_position:
                # 没有持仓，检查入场信号
                # 条件：收盘价突破entry_signal（20日高点）
                if current_price > entry_signal and atr > 0:
                    # 计算头寸规模
                    self.position = self.calculate_position_size(current_price, atr)
                    
                    if self.position > 0:
                        # 入场交易
                        self.entry_price = current_price
                        self.stop_loss = current_price - (self.atr_multiplier * atr)
                        in_position = True
                        
                        self.trades.append({
                            'date': current_date,
                            'action': 'BUY',
                            'price': current_price,
                            'shares': self.position,
                            'stop_loss': self.stop_loss,

                    })

                    self.position = 0
                    in_position = False
                    self.entry_price = 0
                    self.stop_loss = 0
        
        # 最后一天平仓（如果还有持仓）
        if in_position and len(data) > 0:
            last_row = data.iloc[-1]

                'action': 'SELL',

                'shares': self.position,

# 第二步：自定义交易策略类
class TurtleStrategy(bt.Strategy):
    # 第二步之一：定义初始函数
    def __init__(self):
        # 设置均线周期日期长度
        self.n1 = 20        
        # 设置变动幅度
        self.pct = 2
        # 设置self.order的初始值为0，表示处于未交易状态
        self.order = 0
        # 设置买入价格的初始值为0
        self.buy_price = 0
        # 设置卖出价格的初始值为0
        self.sell_price = 0
        # 调用bt指标模块ATR函数，参数传入初始化方法中self.n1变量
        self.atr = bt.indicators.ATR(period=self.n1)
        # 调用bt的indicator指标模块SMA函数，参数传入初始化方法中self.n1变量
        self.sma1 = bt.indicators.SMA(period=self.n1)

    # 第二步之二：定义交易规则。
    def next(self):
        # 如果策略交易状态为真（self.order==1）
        if self.order:
            return
        # 卖出条件1：当前行策略持仓状态为真（self.position==1）
        if self.position:
            # 卖出条件2：当前行的close<self.sell_price（止损）或当前行的close>self.sell_price（止盈）
            if self.data.close < self.sell_price or self.data.close[0] > self.buy_price:
                # 执行bt内部卖出函数sell，卖出量为bt内部postion对象的size属性
                self.sell(size=self.position.size)
        else:
            # 买入条件：当前策略持仓状态为假，且close>self.sma1
            if self.data.close > self.sma1:  
                # 将self.buy_price设为close当前行数值与初始化函数中的self.atr的self.pct倍的加法和
                self.buy_price = self.data.close[0] + self.atr[0] * self.pct
                # 将self.sell_price设为列close当前行数值与初始化函数中的self.atr的self.pct倍50%的减法差
                self.sell_price = self.data.close[0] - self.atr[0] * self.pct*0.5
                # 执行bt内部买入函数buy，买入量由参数size设定为1000股
                self.buy(size=1000,)
